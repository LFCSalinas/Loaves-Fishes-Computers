CREATE TABLE `user` (
  `id` INT UNSIGNED AUTO_INCREMENT,
  `first_name` VARCHAR(25),
  `last_name` VARCHAR(25),
  `email` VARCHAR(50),
  `password` BINARY(60),
  `token` BINARY(20),
  `token_expires` BINARY(13),
  `member_since` VARCHAR(10),
  `status` VARCHAR(8) DEFAULT 'Inactive', 
  `admin` VARCHAR(3) DEFAULT 'No',
  `image` varchar(50) DEFAULT NULL,
  `forms` varchar(50) DEFAULT NULL,
  `motivational_statement` varchar(500) DEFAULT NULL,
  `gender` char(1) DEFAULT NULL,
  `birthday` varchar(10) DEFAULT NULL,
  `place_of_birth` varchar(50) DEFAULT NULL,
  `street` varchar(30) DEFAULT NULL,
  `city` varchar(30) DEFAULT NULL,
  `state` char(2) DEFAULT NULL,
  `phone` varchar(12) DEFAULT NULL,
  `other_email` varchar(50) DEFAULT NULL,
  `emergency_first_name` varchar(20) DEFAULT NULL,
  `emergency_last_name` varchar(20) DEFAULT NULL,
  `emergency_relation` varchar(20) DEFAULT NULL,
  `emergency_phone` varchar(12) DEFAULT NULL,
  `job1_company` varchar(40) DEFAULT NULL,
  `job1_title` varchar(30) DEFAULT NULL,
  `job1_years` varchar(9) DEFAULT NULL,
  `job1_duties` varchar(500) DEFAULT NULL,
  `job2_company` varchar(40) DEFAULT NULL,
  `job2_title` varchar(30) DEFAULT NULL,
  `job2_years` varchar(9) DEFAULT NULL,
  `job2_duties` varchar(500) DEFAULT NULL,
  `skills` varchar(1000) DEFAULT NULL,
  `languages` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC) VISIBLE,
  UNIQUE INDEX `email_UNIQUE` (`email` ASC) VISIBLE
);
